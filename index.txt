    <script>
        // ---------------------------------------------------------------------------------
        // --------------------------------------------------------------------------------- Show poll
        function getFakeData() {
            // const resultString = JSON.stringify(result, null, 2);
            const data = {
                name: "asd",
                uuid: "0000",
                questions: [
                    {
                    uuid: "0001",
                    title: "q1",
                    answers: ["a1", "a2", "a3"]
                    },

                    answers: ["b1", "b2", "b3"]

            console.log('result: ', data);
            return data;
        }
        function createQuestionCardOptions(question) {
            let answersHtml = '<div class="d-flex justify-content-center" role="group" aria-label="Question Options">';
            question.answers.forEach((answer, idx) => {
                answersHtml += `
                    <input type="radio" class="btn-check" name="${question.title}" id="${question.uuid}-${answer}" value="${answer}" autocomplete="off">
                    <label class="btn btn-outline-primary me-2" for="${question.uuid}-${answer}">
                        ${answer}
                    </label>
                `;
            });
            answersHtml += '</div>';
            return answersHtml;
        }
        function createQuestionCard(question) {
            const card = `
                <div class="col-md-4 col-sm-6">
                    <div class="card" style="width: 18rem; margin: 10px;">
                        <div class="card-body text-center">
                            <h5 class="card-title mt-1 mb-4">${question.title}</h5>
                            ${createQuestionCardOptions(question)}
                        </div>
                    </div>
                </div>
            `;
            return card;
        }
        function createQuestionCards(questions) {
            let cards = '';
            questions.forEach(question => {
                cards += createQuestionCard(question);
            });
            $('#result').html(cards);
        }

        function createPollLayout(result) {
            $('#poll-title').html(result.name);
            createQuestionCards(result.questions);
        }
        // ---------------------------------------------------------------------------------
        // --------------------------------------------------------------------------------- Send result
        
        function showAlert(message, alertType) {
            const alertHtml = `
                <div class="alert alert-${alertType} alert-dismissible fade show" role="alert">
                    ${message}
                    <button type="button" class="btn-close" data-bs-dismiss="alert" aria-label="Close"></button>
                </div>
            `;
            $('#alertContainer').html(alertHtml);

            setTimeout(() => {
                $('#alertContainer').html('');
            }, 5000);
        }
        function ensureCreatentialsAreValid(username, password) {
            if (username === '' || password === '') {                
                showAlert('Username and password are required.', 'danger');
                throw new Error('Username and password are required');
            }

        }
        function printSelectedOptions() {
            const result = getFakeData();
            const selectedOptions = result.questions.map((question, index) => {
                const selectedValue = $(`input[name="${question.title}"]:checked`).val();
                return {
                    uuid: question.uuid,
                    title: question.title,
                    value: selectedValue
                };
            });
            const username = $('#username').val();
            const password = $('#password').val();
            ensureCreatentialsAreValid(username, password);
            
            const pollResult = {
                poll: {
                    uuid: result.uuid,
                    user: username,
                    key: password,
                    questions: selectedOptions
                }
            };
            console.log(JSON.stringify(pollResult, null, 2));
        }
        function addSendDataOnClickObserver() {
            $('#printSelections').on('click', printSelectedOptions);
        }
        // ---------------------------------------------------------------------------------
        // --------------------------------------------------------------------------------- Page onload
        $(document).ready(function() {
            console.log('ready!');
            const result = getFakeData();
            createPollLayout(result);
            addSendDataOnClickObserver();
        });
